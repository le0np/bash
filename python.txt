3. Izbrojati Samoglasnike unutar nekog stringa, imate zadano ime funkcije i dolje pozive te funkcije treba dodat samo popunit TODO komentar unutar funkcije:


def count_vowels(s):
    # Pretvara string u mala slova radi jednostavnosti
    s = s.lower()
    # Definira skup samoglasnika
    vowels = {'a', 'e', 'i', 'o', 'u'}
    # Broji samoglasnike u stringu
    count = sum(1 for char in s if char in vowels)
    return count


# Primjer korištenja funkcije
my_string = "Hello World"
num_vowels = count_vowels(my_string)
print("Broj samoglasnika:", num_vowels)
